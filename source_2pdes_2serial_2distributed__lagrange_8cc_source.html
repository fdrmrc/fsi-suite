131  template <int dim, int spacedim>
132  void
133  DistributedLagrange<dim, spacedim>::adjust_grid_refinements(
134  const bool apply_delta_refinement)
135  {
136  namespace bgi = boost::geometry::index;
137  // Now get a vector of all cell centers of the embedded grid and refine
138  // them untill every cell is of diameter smaller than the space
139  // surrounding cells
140  std::vector<
141  std::tuple<Point<spacedim>, decltype(embedded_grid.begin_active())>>
142  centers;
143 
144  for (const auto &cell : embedded_grid.active_cell_iterators())
145  centers.emplace_back(
146  std::make_tuple(embedded_mapping().get_center(cell), cell));
147 
148  // Refine the space grid according to the delta refinement
149  if (apply_delta_refinement && delta_refinement != 0)
150  for (unsigned int i = 0; i < delta_refinement; ++i)
151  {
152  const auto &tree =
153  space_grid_tools_cache
154  ->get_locally_owned_cell_bounding_boxes_rtree();
155  for (const auto &[center, cell] : centers)
156  for (const auto &[space_box, space_cell] :
157  tree | bgi::adaptors::queried(bgi::contains(center)))
158  {
159  space_cell->set_refine_flag();
160  for (const auto face_no : space_cell->face_indices())
161  if (!space_cell->at_boundary(face_no))
162  space_cell->neighbor(face_no)->set_refine_flag();
163  }
164  space_grid.execute_coarsening_and_refinement();
165  }
166 
167  bool done = false;
168  while (done == false)
169  {
170  // Now refine the embedded grid if required
171  const auto &tree = space_grid_tools_cache
172  ->get_locally_owned_cell_bounding_boxes_rtree();
173  // Let's check all cells whose bounding box contains an embedded
174  // center
175  done = true;
176  namespace bgi = boost::geometry::index;
177  for (const auto &[center, cell] : centers)
178  {
179  const auto &[p1, p2] =
180  embedded_mapping().get_bounding_box(cell).get_boundary_points();
181  const auto diameter = p1.distance(p2);
182 
183  for (const auto &[space_box, space_cell] :
184  tree | bgi::adaptors::queried(bgi::contains(center)))
185  if (space_cell->diameter() < diameter)
186  {
187  cell->set_refine_flag();
188  done = false;
189  }
190  }
191  if (done == false)
192  {
193  // Compute again the embedded displacement grid
194  embedded_grid.execute_coarsening_and_refinement();
195  embedded_configuration_dh.distribute_dofs(
196  *embedded_configuration_fe);
197  embedded_configuration.reinit(embedded_configuration_dh.n_dofs());
198  embedded_mapping.initialize(embedded_configuration);
199 
200  // Rebuild the centers vector
201  centers.clear();
202 
203  for (const auto &cell : embedded_grid.active_cell_iterators())
204  centers.emplace_back(
205  std::make_tuple(embedded_mapping().get_center(cell), cell));
206  }
207  }
208 
209  const double embedded_space_maximal_diameter =
210  GridTools::maximal_cell_diameter(embedded_grid, embedded_mapping());
211  const double embedded_space_minimal_diameter =
212  GridTools::minimal_cell_diameter(embedded_grid, embedded_mapping());
213 
214  double space_minimal_diameter =
215  GridTools::minimal_cell_diameter(space_grid);
216  double space_maximal_diameter =
217  GridTools::maximal_cell_diameter(space_grid);
218 
219  deallog << "Space min/max diameters: " << space_minimal_diameter << "/"
220  << space_maximal_diameter << std::endl
221  << "Embedded space min/max diameters: "
222  << embedded_space_minimal_diameter << "/"
223  << embedded_space_maximal_diameter << std::endl;
224  }

228  void
229  DistributedLagrange<dim, spacedim>::setup_dofs()
230  {
231  TimerOutput::Scope timer_section(monitor, "setup_dofs");
232 
233  space_dh.distribute_dofs(*space_fe);
234  constraints.clear();
235  DoFTools::make_hanging_node_constraints(space_dh, constraints);
236  boundary_conditions.apply_essential_boundary_conditions(*space_mapping,
237  space_dh,
238  constraints);
239  constraints.close();
240 
241  DynamicSparsityPattern dsp(space_dh.n_dofs(), space_dh.n_dofs());
242  DoFTools::make_sparsity_pattern(space_dh, dsp, constraints);
243  stiffness_sparsity.copy_from(dsp);
244  stiffness_matrix.reinit(stiffness_sparsity);
245  solution.reinit(space_dh.n_dofs());
246  rhs.reinit(space_dh.n_dofs());
247  deallog << "Embedding dofs: " << space_dh.n_dofs() << std::endl;
248 
249  embedded_dh.distribute_dofs(*embedded_fe);
250  lambda.reinit(embedded_dh.n_dofs());
251  embedded_rhs.reinit(embedded_dh.n_dofs());
252  embedded_value.reinit(embedded_dh.n_dofs());
253  deallog << "Embedded dofs: " << embedded_dh.n_dofs() << std::endl;
254 
255  boundary_conditions.apply_natural_boundary_conditions(
256  *space_mapping, space_dh, constraints, stiffness_matrix, rhs);

286  void
287  DistributedLagrange<dim, spacedim>::assemble_system()
288  {
289  const auto space_quad = ParsedTools::Components::get_cell_quadrature(
290  space_grid, space_fe->tensor_degree() + 1);
291  const auto embedded_quad = ParsedTools::Components::get_cell_quadrature(
292  embedded_grid, embedded_fe->tensor_degree() + 1);
293  {
294  TimerOutput::Scope timer_section(monitor, "Assemble system");
295  // Stiffness matrix and rhs
296  MatrixTools::create_laplace_matrix(
297  space_dh,
298  space_quad,
299  stiffness_matrix,
300  forcing_term,
301  rhs,
302  static_cast<const Function<spacedim> *>(nullptr),
303  constraints);
304  }
305  {
306  TimerOutput::Scope timer_section(monitor, "Assemble coupling system");
307  NonMatching::create_coupling_mass_matrix(*space_grid_tools_cache,
308  space_dh,
309  embedded_dh,
310  embedded_quad,
311  coupling_matrix,
312  constraints,
313  ComponentMask(),
314  ComponentMask(),
315  embedded_mapping(),
316  embedded_constraints);
317 
318  // The rhs of the Lagrange multiplier as a function to plot
319  VectorTools::interpolate(embedded_mapping(),
320  embedded_dh,
321  embedded_value_function,
322  embedded_value);
323 
324  // The rhs of the Lagrange multiplier
325  VectorTools::create_right_hand_side(embedded_mapping(),
326  embedded_dh,
327  embedded_quad,
328  embedded_value_function,
329  embedded_rhs);

333  void
334  DistributedLagrange<dim, spacedim>::solve()
335  {
336  TimerOutput::Scope timer_section(monitor, "Solve system");
337 
338  auto A = linear_operator(stiffness_matrix);
339  auto Bt = linear_operator(coupling_matrix);
340  auto B = transpose_operator(Bt);
341  auto A_inv = A;
342  SparseDirectUMFPACK A_inv_direct;
343 
344  // auto control = stiffness_inverse_operator.setup_new_solver_control();
345  // TrilinosWrappers::SolverDirect solver_direct(
346  // *control,
347  // TrilinosWrappers::SolverDirect::AdditionalData(true,
348  // "Amesos_Pardiso"));
349 
350  if (use_direct_solver)
351  {
352  A_inv_direct.initialize(stiffness_matrix);
353  A_inv = linear_operator(A, A_inv_direct);
354 
355  // solver_direct.initialize(stiffness_matrix);
356  // A_inv = A;
357  // A_inv.vmult = [&](Vector<double> &dst, const Vector<double> &src) {
358  // solver_direct.solve(dst, src);
359  // };
360  }
361  else
362  {
363  stiffness_preconditioner.initialize(stiffness_matrix);
364  A_inv = stiffness_inverse_operator(A, stiffness_preconditioner);
365  }
366 
367 
368 
369  auto S = B * A_inv * Bt;
370  auto S_prec = identity_operator(S);
371  auto S_inv = schur_inverse_operator(S, S_prec);
372 
373  lambda = S_inv * (B * A_inv * rhs - embedded_rhs);
374  solution = A_inv * (rhs - Bt * lambda);
375  constraints.distribute(solution);
376  }

404  template <int dim, int spacedim>
405  void
406  DistributedLagrange<dim, spacedim>::run()
407  {
408  deallog.depth_console(console_level);
409  generate_grids_and_fes();
410  for (const auto &cycle : grid_refinement.get_refinement_cycles())
411  {
412  deallog.push("Cycle " + Utilities::int_to_string(cycle));
413  setup_dofs();
414  setup_coupling();
415  assemble_system();
416  solve();
417  error_table.error_from_exact(space_dh, solution, exact_solution);
418  output_results(cycle);
419  if (cycle < grid_refinement.get_n_refinement_cycles() - 1)
420  {
421  grid_refinement.estimate_mark_refine(space_dh,
422  solution,
423  space_grid);
424  adjust_grid_refinements(false);
425  }
426  deallog.pop();
427  }
428  error_table.output_table(std::cout);
429  }
