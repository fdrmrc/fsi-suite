81  template <int dim, int spacedim, typename LacType>
82  void
83  DistributedLagrange<dim, spacedim, LacType>::assemble_system()
84  {
85  const auto space_quad = ParsedTools::Components::get_cell_quadrature(
86  space.triangulation, space.finite_element().tensor_degree() + 1);
87  const auto embedded_quad = ParsedTools::Components::get_cell_quadrature(
88  embedded.triangulation, embedded.finite_element().tensor_degree() + 1);
89  {
90  TimerOutput::Scope timer_section(space.timer, "Assemble system");
91  // Stiffness matrix and rhs
92  MatrixTools::create_laplace_matrix(
93  space.dof_handler,
94  space_quad,
95  space.matrix.block(0, 0),
96  space.forcing_term,
97  space.rhs.block(0),
98  static_cast<const Function<spacedim> *>(nullptr),
99  space.constraints);
100  }
101  {
102  TimerOutput::Scope timer_section(space.timer,
103  "Assemble coupling system");
104  coupling.assemble_matrix(coupling_matrix);
105  }
106  {
107  TimerOutput::Scope timer_section(space.timer, "Assemble embedded mass");
108  MatrixCreator::create_mass_matrix(
109  embedded.dof_handler,
110  embedded_quad,
111  embedded.matrix,
112  static_cast<const Function<spacedim> *>(nullptr),
113  embedded.constraints);
114 
115  // The rhs of the Lagrange multiplier as a function to plot
116  VectorTools::interpolate(embedded.dof_handler,
117  embedded.forcing_term,
118  embedded.solution);
119 
120  // The rhs of the Lagrange multiplier
121  VectorTools::create_right_hand_side(embedded.dof_handler,
122  embedded_quad,
123  embedded.forcing_term,
124  embedded.rhs);
125  }
126  }

130  template <int dim, int spacedim, typename LacType>
131  void
132  DistributedLagrange<dim, spacedim, LacType>::solve()
133  {
134  TimerOutput::Scope timer_section(space.timer, "Solve system");
135 
136  auto A = linear_operator(space.matrix.block(0, 0));
137  auto Bt = linear_operator(coupling_matrix);
138  auto B = transpose_operator(Bt);
139  auto A_inv = A;
140  auto M = linear_operator(embedded.matrix.block(0, 0));
141  auto M_inv = M;
142 
143  typename LacType::DirectSolver A_inv_direct;
144  typename LacType::DirectSolver M_inv_direct;
145 
146  if (use_direct_solver)
147  {
148  A_inv_direct.initialize(space.matrix.block(0, 0));
149  A_inv = linear_operator(A, A_inv_direct);
150  M_inv_direct.initialize(embedded.matrix.block(0, 0));
151  M_inv = linear_operator(M, M_inv_direct);
152  }
153  else
154  {
155  space.preconditioner.initialize(space.matrix.block(0, 0));
156  A_inv = space.inverse_operator(A, space.preconditioner);
157  }
158 
159 
160  auto &lambda = embedded.solution.block(0);
161  auto &embedded_rhs = embedded.rhs.block(0);
162  auto &solution = space.solution.block(0);
163  auto &rhs = space.rhs.block(0);
164 
165  if (false)
166  {
167  // Solve the system
168  solution = A_inv * rhs;
169  lambda = M_inv * B * solution;
170  }
171  else
172  {
173  auto S = B * A_inv * Bt;
174  auto S_prec = identity_operator(S);
175  auto S_inv = embedded.inverse_operator(S, M_inv);
176  lambda = S_inv * (B * A_inv * rhs - embedded_rhs);
177  solution = A_inv * (rhs - Bt * lambda);
178  }
179  embedded.constraints.distribute(lambda);
180  embedded.locally_relevant_solution = embedded.solution;
181  space.constraints.distribute(solution);
182  space.locally_relevant_solution = space.solution;
183  }

198  template <int dim, int spacedim, typename LacType>
199  void
200  DistributedLagrange<dim, spacedim, LacType>::run()
201  {
202  deallog.depth_console(space.verbosity_level);
203  generate_grids();
204  for (const auto &cycle : space.grid_refinement.get_refinement_cycles())
205  {
206  deallog.push("Cycle " + Utilities::int_to_string(cycle));
207  setup_system();
208  assemble_system();
209  solve();
210  space.estimate(space.error_per_cell);
211  embedded.estimate(embedded.error_per_cell);
212  output_results(cycle);
213  if (cycle < space.grid_refinement.get_n_refinement_cycles() - 1)
214  {
215  space.mark(space.error_per_cell);
216  space.refine();
217  coupling.adjust_grid_refinements(space.triangulation,
218  embedded.triangulation,
219  false);
220  }
221  deallog.pop();
222  }
223  if (space.mpi_rank == 0)
224  {
225  space.error_table.output_table(std::cout);
226  embedded.error_table.output_table(std::cout);
227  }
228  }
