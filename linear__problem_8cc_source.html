87  template <int dim, int spacedim>
88  void
89  LinearProblem<dim, spacedim>::setup_system()
90  {
91  TimerOutput::Scope timer_section(timer, "setup_system");
92  pcout << "System setup" << std::endl;
93  const auto ref_cells = triangulation.get_reference_cells();
94  AssertThrow(
95  ref_cells.size() == 1,
96  ExcMessage(
97  "This program does nots support mixed simplx/hex grid types."));
98 
99  // Compatible FE space and grid.
100  AssertThrow(finite_element().reference_cell() == ref_cells[0],
101  ExcMessage("The finite element must be defined on the same "
102  "cell type as the grid."));
103 
104  dof_handler.distribute_dofs(finite_element);
105 
106  // Since our code runs both for simplex grids and for hyper-cube grids, we
107  // need to make sure that we build the correct mapping for the grid. In
108  // this code we actually use a linear mapping, independently on the order
109  // of the finite element space.
110  mapping = get_default_linear_mapping(triangulation).clone();
111  pcout << "Number of dofs " << dof_handler.n_dofs() << std::endl;
112 
113  constraints.clear();
114  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
115 
116  // We now check that the boundary conditions are consistent with the
117  // triangulation object, that is, we check that the boundary indicators
118  // specified in the parameter file are actually present in the
119  // triangulation, that no boundary indicator is specified twice, and that
120  // all boundary ids of the triangulation are actually covered by the
121  // parameter file configuration.
122  boundary_conditions.check_consistency(triangulation);
123 
124  // This is where we apply essential boundary conditions. The
125  // ParsedTools::BoundaryConditions class takes care of collecting boundary
126  // ids, and calling the appropriate function to apply the boundary
127  // condition on the selected boundary ids. Essential boundary conditions
128  // need to be incorporated in the constraints of the linear system, since
129  // they are part of the definition of the solution space.
130  //
131  // Natural bondary conditions, on the other hand, need access to the rhs
132  // of the problem and are not treated via constraints. We will deal with
133  // them later on.
134  boundary_conditions.apply_essential_boundary_conditions(dof_handler,
135  constraints);
136 
137  // If necessary, derived functions can add constraints to the system here.
138  add_constraints_call_back();
139  constraints.close();
140 
141  const auto blocks =
142  ParsedTools::Components::block_indices(component_names,
143  component_names);
144  // renumber dofs in a blockwise manner.
145  DoFRenumbering::component_wise(dof_handler, blocks);
146  dofs_per_block = DoFTools::count_dofs_per_fe_block(dof_handler, blocks);
147 
148  locally_owned_dofs =
149  dof_handler.locally_owned_dofs().split_by_block(dofs_per_block);
150 
151  IndexSet non_blocked_locally_relevant_dofs;
152  DoFTools::extract_locally_relevant_dofs(
153  dof_handler, non_blocked_locally_relevant_dofs);
154  locally_relevant_dofs =
155  non_blocked_locally_relevant_dofs.split_by_block(dofs_per_block);
156 
157  pcout << "Number of degrees of freedom: " << dof_handler.n_dofs() << " ("
158  << Patterns::Tools::to_string(dofs_per_block) << ")" << std::endl;
159 
160  constraints.clear();
161  constraints.reinit(non_blocked_locally_relevant_dofs);
162  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
163 
164  boundary_conditions.apply_essential_boundary_conditions(*mapping,
165  dof_handler,
166  constraints);
167  constraints.close();
168 
169  // BlockDynamicSparsityPattern dsp(dofs_per_block, dofs_per_block);
170  TrilinosWrappers::BlockSparsityPattern dsp(locally_owned_dofs,
171  locally_owned_dofs,
172  locally_relevant_dofs,
173  mpi_communicator);
174 
175  DoFTools::make_sparsity_pattern(dof_handler,
176  dsp,
177  constraints,
178  false,
179  Utilities::MPI::this_mpi_process(
180  mpi_communicator));
181 
182  // SparsityTools::distribute_sparsity_pattern(dsp,
183  // locally_owned_dofs,
184  // mpi_communicator,
185  // locally_relevant_dofs);
186  dsp.compress();
187 
188  system_block_matrix.reinit(dsp);
189 
190  block_solution.reinit(locally_owned_dofs, mpi_communicator);
191  system_block_rhs.reinit(locally_owned_dofs, mpi_communicator);
192 
193  locally_relevant_block_solution.reinit(locally_owned_dofs,
194  locally_relevant_dofs,
195  mpi_communicator);
196 
197  error_per_cell.reinit(triangulation.n_active_cells());
198 
199  boundary_conditions.apply_natural_boundary_conditions(*mapping,
200  dof_handler,
201  constraints,
202  system_block_matrix,
203  system_block_rhs);
204  // Update functions with standard constants
205  exact_solution.update_constants({});
206  forcing_term.update_constants({});
207 
208  // Now call anything else that may be needed from the user side
209  setup_system_call_back();
210  }

239  template <int dim, int spacedim>
240  void
241  LinearProblem<dim, spacedim>::assemble_system()
242  {
243  TimerOutput::Scope timer_section(timer, "assemble_system");
244  Quadrature<dim> quadrature_formula =
245  ParsedTools::Components::get_cell_quadrature(
246  triangulation, finite_element().tensor_degree() + 1);
247 
248  Quadrature<dim - 1> face_quadrature_formula =
249  ParsedTools::Components::get_face_quadrature(
250  triangulation, finite_element().tensor_degree() + 1);
251 
252  ScratchData scratch(*mapping,
253  finite_element(),
254  quadrature_formula,
255  update_values | update_gradients |
256  update_quadrature_points | update_JxW_values,
257  face_quadrature_formula,
258  update_values | update_quadrature_points |
259  update_JxW_values);
260 
261  CopyData copy(finite_element().n_dofs_per_cell());
262 
263  auto worker = [&](const auto &cell, auto &scratch, auto &copy) {
264  assemble_system_one_cell(cell, scratch, copy);
265  };
266 
267  auto copier = [&](const auto &copy) { copy_one_cell(copy); };
268 
269  using CellFilter = FilteredIterator<
270  typename DoFHandler<dim, spacedim>::active_cell_iterator>;
271 
272  WorkStream::run(CellFilter(IteratorFilters::LocallyOwnedCell(),
273  dof_handler.begin_active()),
274  CellFilter(IteratorFilters::LocallyOwnedCell(),
275  dof_handler.end()),
276  worker,
277  copier,
278  scratch,
279  copy);
280 
281  system_block_matrix.compress(VectorOperation::add);
282  system_block_rhs.compress(VectorOperation::add);
283 
284  assemble_system_call_back();
285  }

378  template <int dim, int spacedim>
379  void
380  LinearProblem<dim, spacedim>::run()
381  {
382  print_system_info();
383  grid_generator.generate(triangulation);
384  for (const auto &cycle : grid_refinement.get_refinement_cycles())
385  {
386  pcout << "Cycle " << cycle << std::endl;
387  setup_system();
388  assemble_system();
389  solve();
390  estimate(error_per_cell);
391  output_results(cycle);
392  if (cycle < grid_refinement.get_n_refinement_cycles() - 1)
393  {
394  mark(error_per_cell);
395  refine();
396  }
397  }
398  if (pcout.is_active())
399  error_table.output_table(std::cout);
400  }
